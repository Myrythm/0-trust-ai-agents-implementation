# Feature 5: `zta.tools` Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add `zta/tools.py` — `@tool` decorator + `ToolRegistry`. The decorator marks a function as a tool and stamps it with a name; the registry stores named tools and looks them up. This module **does not** enforce policy; that's F6's job. F5 is just the marker + lookup mechanism.

**Architecture:** A tiny module. The `@tool` decorator is a thin wrapper that uses `functools.wraps` and stashes a `__zta_tool_name__` attribute on the wrapper. `ToolRegistry` is a thin in-process mapping. Single-process; no thread-safety (MVP — runtime is single-threaded).

**Tech Stack:** stdlib only (`functools`, `collections.abc`).

## Contract (per spec section 3.4)

```python
def tool(name: str | None = None) -> Callable[[Callable], Callable]: ...

class ToolRegistry:
    def __init__(self) -> None: ...
    def register(self, fn: Callable, *, name: str | None = None) -> None: ...
    def get(self, name: str) -> Callable: ...            # raises ToolError if missing
    def list(self) -> list[str]: ...                     # sorted
```

**Naming rules:**
- If the user passes an explicit `name=` (or `@tool("name")`), use it
- Otherwise, if the function was decorated with `@tool`, use its `__zta_tool_name__`
- Otherwise, use `fn.__name__`

**Failure modes:**
- `register` with the same name twice → silently overwrites (last wins)
- `get` with an unknown name → `ToolError` (not `KeyError`; deny-by-default surface)
- `list()` returns names in sorted order (deterministic for UIs)

**Thread-safety:** Not thread-safe. The MVP runtime is single-threaded; if a tool must be called from multiple threads, use a `threading.Lock` around the registry at the call site.

import pytest

from zta.errors import ToolError, ZTAError

def test_tool_decorator_marks_function() -> None:
    @tool
    def f(x: int) -> int:
        return x * 2

    assert getattr(f, "__zta_tool_name__") == "f"

    @tool("custom_name")
    def f(x: int) -> int:
        return x * 2

    assert f.__zta_tool_name__ == "custom_name"

    @tool
    def add(a: int, b: int) -> int:
        return a + b

    assert add(2, 3) == 5


def test_tool_decorator_preserves_metadata() -> None:
    @tool
    def my_func() -> int:
        """docstring here"""
        return 42

    assert my_func.__name__ == "my_func"
    assert my_func.__doc__ == "docstring here"


# ---------- ToolRegistry ----------


def test_registry_register_and_get() -> None:
    reg = ToolRegistry()

    @tool
    def hello(name: str) -> str:
        return f"hello {name}"

    reg.register(hello)
    assert reg.get("hello") is hello


def test_registry_register_with_explicit_name() -> None:
    reg = ToolRegistry()

    def fn() -> str:
        return "x"

    reg.register(fn, name="custom")
    assert reg.get("custom") is fn


def test_registry_register_twice_overwrites() -> None:
    reg = ToolRegistry()

    def a() -> str:
        return "a"

    def b() -> str:
        return "b"

    reg.register(a, name="f")
    reg.register(b, name="f")
    assert reg.get("f") is b


def test_registry_get_missing_raises_tool_error() -> None:
    reg = ToolRegistry()
    with pytest.raises(ToolError):
        reg.get("ghost")


def test_registry_list_returns_sorted_names() -> None:
    reg = ToolRegistry()

    @tool
    def zebra() -> None:
        return None

    @tool
    def apple() -> None:
        return None

    @tool
    def mango() -> None:
        return None

    reg.register(zebra)
    reg.register(apple)
    reg.register(mango)
    assert reg.list() == ["apple", "mango", "zebra"]


def test_registry_register_picks_up_decorator_name() -> None:
    reg = ToolRegistry()

    @tool("db_query")
    def q(sql: str) -> str:
        return sql

    reg.register(q)
    # registered under the decorator's name, not the function's __name__
    assert reg.get("db_query") is q

```python
"""Tool marker + registry.

`@tool` decorates a function to mark it as a ZTA tool and stamps the
chosen name on the wrapper. `ToolRegistry` is a thin in-process map for
storing and looking up tools by name. This module is the marker +
lookup; policy enforcement lives in `zta.runtime` (F6).
"""

from __future__ import annotations

from collections.abc import Callable
from functools import wraps
from typing import Any, TypeVar

from zta.errors import ToolError

F = TypeVar("F", bound=Callable[..., Any])

_TOOL_NAME_ATTR = "__zta_tool_name__"


def tool(name: str | None = None) -> Callable[[F], F]:
    """Mark a function as a ZTA tool.

    Usage:
        @tool                 # name = fn.__name__
        def db_query(sql): ...

        @tool("db.query")     # explicit name
        def db_query(sql): ...
    """

    def decorator(fn: F) -> F:
        tool_name = name if name is not None else fn.__name__

        @wraps(fn)
        def wrapper(*args: Any, **kwargs: Any) -> Any:
            return fn(*args, **kwargs)

        setattr(wrapper, _TOOL_NAME_ATTR, tool_name)
        return wrapper  # type: ignore[return-value]

    return decorator


class ToolRegistry:
    """In-process map of tool name → callable."""

    def __init__(self) -> None:
        self._tools: dict[str, Callable[..., Any]] = {}

    def register(self, fn: Callable[..., Any], *, name: str | None = None) -> None:
        """Register `fn` under the chosen name (or its tool/function name)."""
        chosen = (
            name
            if name is not None
            else getattr(fn, _TOOL_NAME_ATTR, None) or fn.__name__
        )
        self._tools[chosen] = fn

    def get(self, name: str) -> Callable[..., Any]:
        """Return the tool registered as `name`; raise ToolError if missing."""
        try:
            return self._tools[name]
        except KeyError as exc:
            raise ToolError(f"tool not registered: {name!r}") from exc

    def list(self) -> list[str]:
        """Return the registered tool names in sorted order."""
        return sorted(self._tools)
```
